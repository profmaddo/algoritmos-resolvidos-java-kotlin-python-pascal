```java
import java.util.Scanner;

public class EstatisticasNumeros {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int num, soma = 0, qtd = 0, pares = 0, somaPares = 0, impares = 0;
        int maior = Integer.MIN_VALUE, menor = Integer.MAX_VALUE;

        System.out.println("Digite n√∫meros inteiros (100000 encerra):");

        while (true) {
            System.out.print("N√∫mero: ");
            num = scanner.nextInt();
            if (num == 100000) break;

            soma += num;
            qtd++;

            if (num > maior) maior = num;
            if (num < menor) menor = num;

            if (num % 2 == 0) {
                pares++;
                somaPares += num;
            } else {
                impares++;
            }
        }

        if (qtd > 0) {
            double media = (double) soma / qtd;
            System.out.println("Soma: " + soma);
            System.out.println("Quantidade: " + qtd);
            System.out.printf("M√©dia: %.2f%n", media);
            System.out.println("Maior n√∫mero: " + maior);
            System.out.println("Menor n√∫mero: " + menor);

            if (pares > 0) {
                double mediaPares = (double) somaPares / pares;
                System.out.printf("M√©dia dos pares: %.2f%n", mediaPares);
            } else {
                System.out.println("Nenhum n√∫mero par digitado.");
            }

            double porcentagemImpares = (impares * 100.0) / qtd;
            System.out.printf("Porcentagem de √≠mpares: %.2f%%%n", porcentagemImpares);
        } else {
            System.out.println("Nenhum n√∫mero v√°lido digitado.");
        }
    }
}
```
